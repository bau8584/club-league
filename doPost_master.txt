/**
 * 초등 배드민턴 리그 마스터 인증 및 교사 제어 API 엔진 (이메일 자가 복구 & Concurrency Lock 통합 버전)
 * 마스터 스프레드시트 ID: 1EcIVyYIWvc9OLf5YwF3XCazGy1brcZ53aE_btpdvHr4
 */

const MASTER_SPREADSHEET_ID = "1EcIVyYIWvc9OLf5YwF3XCazGy1brcZ53aE_btpdvHr4";

// 1. GET 요청 처리 (교사 목록 조회 등 - 최고 관리자용)
function doGet(e) {
  try {
    const action = e.parameter.action;
    const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
    const userSheet = ss.getSheetByName("users") || ss.getSheets()[0];

    if (action === "GET_TEACHERS") {
      const rows = userSheet.getDataRange().getValues();
      const teachers = [];
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          const row = rows[i];
          if (!row[0]) continue;
          // 최고 관리자 계정은 노출하지 않고 교사/학생 계정만 반환
          if (row[2] === "TEACHER" || row[2] === "STUDENT") {
            teachers.push({
              loginId: String(row[0]),
              role: String(row[2]),
              schoolName: String(row[3]),
              userName: String(row[4]),
              scriptUrl: String(row[5]),
              createdAt: String(row[6]),
              email: row[7] ? String(row[7]) : ""
            });
          }
        }
      }
      return ContentService.createTextOutput(JSON.stringify({ status: "success", teachers: teachers }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "알 수 없는 GET Action입니다." }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// 2. POST 요청 처리 (로그인, 회원가입, 회원 관리 제어)
function doPost(e) {
  // 동시성 제어를 위한 스크립트 락 획득
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(5000); // 5초 동안 락 대기
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: "다른 사용자가 로그인/등록 중입니다 (서버 혼잡). 3초 후 다시 시도해주세요."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  try {
    const requestData = JSON.parse(e.postData.contents);
    const action = requestData.action;
    const ss = SpreadsheetApp.openById(MASTER_SPREADSHEET_ID);
    const userSheet = ss.getSheetByName("users") || ss.getSheets()[0];

    // 컬럼 H(8번째 열)에 email 헤더가 없는 경우 자동 생성
    var maxCols = userSheet.getLastColumn();
    if (maxCols < 8) {
      userSheet.getRange(1, 8).setValue("email");
    }

    if (action === "REGISTER") {
      const { loginId, password, role, schoolName, userName, scriptUrl, email } = requestData;
      const rows = userSheet.getDataRange().getValues();
      
      // 중복 체크
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          if (String(rows[i][0]).trim().toLowerCase() === String(loginId).trim().toLowerCase()) {
            lock.releaseLock();
            return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "이미 가입 완료된 학교(로그인 ID)입니다." }))
              .setMimeType(ContentService.MimeType.JSON);
          }
        }
      }

      // 등록 실행
      userSheet.appendRow([
        String(loginId).trim(),
        String(password).trim(),
        String(role).trim(),
        String(schoolName).trim(),
        String(userName).trim(),
        scriptUrl ? String(scriptUrl).trim() : "",
        new Date().toISOString(),
        email ? String(email).trim() : ""
      ]);

      lock.releaseLock();
      return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "정상적으로 등록되었습니다." }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (action === "LOGIN") {
      const { loginId, password, role } = requestData;
      const rows = userSheet.getDataRange().getValues();
      
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          const row = rows[i];
          const dbLoginId = String(row[0]).trim().toLowerCase();
          const dbPassword = String(row[1]).trim();
          const dbRole = String(row[2]).trim();

          if (dbLoginId === String(loginId).trim().toLowerCase() && 
              dbPassword === String(password).trim() && 
              dbRole === String(role).trim()) {
            
            lock.releaseLock();
            return ContentService.createTextOutput(JSON.stringify({
              status: "success",
              user: {
                loginId: String(row[0]),
                role: String(row[2]),
                schoolName: String(row[3]),
                userName: String(row[4]),
                scriptUrl: String(row[5]),
                createdAt: String(row[6]),
                email: row[7] ? String(row[7]) : ""
              }
            })).setMimeType(ContentService.MimeType.JSON);
          }
        }
      }

      lock.releaseLock();
      return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "로그인 정보(ID 또는 인증코드)가 일치하지 않습니다." }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (action === "RECOVER_PASSWORD") {
      const { schoolName, email } = requestData;
      const rows = userSheet.getDataRange().getValues();
      
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          const row = rows[i];
          const dbSchool = String(row[3]).trim().toLowerCase();
          const dbEmail = row[7] ? String(row[7]).trim().toLowerCase() : "";

          if (dbSchool === String(schoolName).trim().toLowerCase() && 
              dbEmail === String(email).trim().toLowerCase()) {
            
            const password = String(row[1]);
            const userName = String(row[4]);
            
            // 이메일 전송
            try {
              sendRecoverEmail(email, schoolName, userName, password);
            } catch (mailErr) {
              lock.releaseLock();
              return ContentService.createTextOutput(JSON.stringify({ 
                status: "error", 
                message: "메일 발송에 실패했습니다. 관리자에게 문의하세요. (오류: " + mailErr.toString() + ")" 
              })).setMimeType(ContentService.MimeType.JSON);
            }

            lock.releaseLock();
            return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "비밀번호가 등록된 이메일로 발송되었습니다." }))
              .setMimeType(ContentService.MimeType.JSON);
          }
        }
      }

      lock.releaseLock();
      return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "입력하신 학교명과 이메일 정보와 일치하는 계정을 찾을 수 없습니다." }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (action === "DELETE_TEACHER") {
      const { targetId } = requestData;
      const rows = userSheet.getDataRange().getValues();
      
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          if (String(rows[i][0]).trim().toLowerCase() === String(targetId).trim().toLowerCase()) {
            userSheet.deleteRow(i + 1);
            lock.releaseLock();
            return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "계정이 정상적으로 삭제되었습니다." }))
              .setMimeType(ContentService.MimeType.JSON);
          }
        }
      }
      lock.releaseLock();
      return ContentService.createTextOutput(JSON.stringify({ status: "fail", message: "삭제할 계정을 찾지 못했습니다." }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (action === "UPDATE_TEACHER") {
      const { targetId, nextSchoolName, nextUserName, nextScriptUrl } = requestData;
      const rows = userSheet.getDataRange().getValues();
      
      if (rows.length > 1) {
        for (let i = 1; i < rows.length; i++) {
          if (String(rows[i][0]).trim().toLowerCase() === String(targetId).trim().toLowerCase()) {
            const rowNum = i + 1;
            if (nextSchoolName) userSheet.getRange(rowNum, 4).setValue(nextSchoolName);
            if (nextUserName) userSheet.getRange(rowNum, 5).setValue(nextUserName);
            if (nextScriptUrl !== undefined) userSheet.getRange(rowNum, 6).setValue(nextScriptUrl);
            
            lock.releaseLock();
            return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "계정 정보가 성공적으로 변경되었습니다." }))
              .setMimeType(ContentService.MimeType.JSON);
          }
        }
      }
      lock.releaseLock();
      return ContentService.createTextOutput(JSON.stringify({ status: "fail", message: "수정할 계정을 찾지 못했습니다." }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    lock.releaseLock();
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "알 수 없는 POST Action입니다." }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    lock.releaseLock();
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
